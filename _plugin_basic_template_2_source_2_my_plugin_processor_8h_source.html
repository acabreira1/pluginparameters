130  void runAfterParamChange(int paramIndex,UpdateFromFlags updateFromFlag){
131  //Generally you don't want to run any updates when you are loading from XML
132  //because it is often more efficient to initialize everything all at once when
133  //all the parameter values have been loaded by overriding
134  //updateProcessorHostAndUiFromXml(...) (see below).
135  if (updateFromFlag&UPDATE_FROM_XML) return;
136 
137  switch(paramIndex){
138  case floatIndex: {
139  logVar=floatVar;
140  getParam(logIndex)->updateHostAndUi(false,UPDATE_FROM_PROCESSOR);
141  logWith0Var=floatVar;
142  getParam(logWith0Index)->updateHostAndUi(false,UPDATE_FROM_PROCESSOR);
143  symSignedLogVar=floatVar;
144  getParam(symSignedLogIndex)->updateHostAndUi(false,UPDATE_FROM_PROCESSOR);
145  asymSignedLogVar=floatVar;
146  getParam(asymSignedLogIndex)->updateHostAndUi(false,UPDATE_FROM_PROCESSOR);
147  break;
148  }
149  case logIndex: {
150  floatVar=logVar;
151  getParam(floatIndex)->updateHostAndUi(false,UPDATE_FROM_PROCESSOR);
152  break;
153  }
154  case logWith0Index: {
155  floatVar=logWith0Var;
156  getParam(floatIndex)->updateHostAndUi(false,UPDATE_FROM_PROCESSOR);
157  break;
158  }
159  case symSignedLogIndex: {
160  floatVar=symSignedLogVar;
161  getParam(floatIndex)->updateHostAndUi(false,UPDATE_FROM_PROCESSOR);
162  break;
163  }
164  case asymSignedLogIndex: {
165  floatVar=asymSignedLogVar;
166  getParam(floatIndex)->updateHostAndUi(false,UPDATE_FROM_PROCESSOR);
167  break;
168  }
169  case boolButtonIndex: {
170  AlertWindow::showMessageBox(AlertWindow::WarningIcon,"Warning!","Aaaaaah!!");
171  break;
172  }
173  default: break;
174  }
