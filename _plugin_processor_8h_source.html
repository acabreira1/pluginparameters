43  void mapGlobalIndex(ParamGroup *root, bool countIfAutomate, ParamGroup **map,int *indexMap) {
44  for (int i=0;i<root->getNumParams();i++){
45  Param* param=root->getParam(i);
46  if (param->automationIsOn()==countIfAutomate){
47  jassert(param->getGlobalIndex()<((countIfAutomate)?getNumAutomatedParams():getNumNonAutomatedParams()));
48  map[param->getGlobalIndex()]=root;
49  indexMap[param->getGlobalIndex()]=i;
50  }
51  }
52  for (int g=0;g<root->getNumParamGroups();g++)
53  mapGlobalIndex(root->getParamGroup(g),countIfAutomate,map,indexMap);
54  }

85  void updateHostAndUi(Param *const param, float newValue,bool runAfterParamChange=true,bool updateUi=true){
86  const int globalIndex=param->getGlobalIndex();
87  ParamGroup *localParamGroup;
88  int paramIndex;
89  if (param->automationIsOn()){ //if automated, notify host
90  //and get corresponding localParamGroup and paramIndex
91  //notify host
92  sendParamChangeMessageToListeners (globalIndex, newValue);
93  localParamGroup=groupAutomated[globalIndex];
94  paramIndex=indexInGroupAutomated[globalIndex];
95 
96  } else { //if not automated skip host notification
97  // but get corresponding localParamGroup and paramIndex
98  localParamGroup=groupNonAutomated[globalIndex];
99  paramIndex=indexInGroupNonAutomated[globalIndex];
100  }
101 
102  if (runAfterParamChange){
103  //"runAfterParamChange" defined in its ParamGroup
104  localParamGroup->runAfterParamChange(paramIndex,param->getUpdateFromFlag());
105  //"runAfterParamGroupChange" defined in its parent ParamGroup
106  if (localParamGroup->getParentParamGroup()!=nullptr)
107  localParamGroup->getParentParamGroup()->runAfterParamGroupChange(localParamGroup->getIndex(),paramIndex,param->getUpdateFromFlag());
108  }
109  if (updateUi){
110  param->updateUi(true);
111  }
112 
113  }

152  //automated parameters:
